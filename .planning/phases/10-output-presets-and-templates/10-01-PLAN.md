       // Resolve --preset to JQ expression.
       if preset != "" {
           if jqFilter != "" {
               apiErr := &cferrors.APIError{

               }

           jqFilter = expr
       }

       ```go
       func availablePresets(p config.Profile) string {
           if len(p.Presets) == 0 {
               return "(none)"
           }
           names := make([]string, 0, len(p.Presets))
           for k := range p.Presets {
               names = append(names, k)
           }
           sort.Strings(names)
           return strings.Join(names, ", ")
       }
       ```
