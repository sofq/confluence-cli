```go
// ExtractResource extracts the resource name from a Confluence v2 path.
// Confluence paths start at /{resource}/... with no version prefix.
// For /pages/{id} → "pages"
// For /spaces/{id}/role-assignments → "spaces"
// Fallback: first non-param, non-empty segment.
func ExtractResource(path string) string {
    segments := strings.Split(strings.TrimPrefix(path, "/"), "/")
    for _, s := range segments {
        if s != "" && !strings.HasPrefix(s, "{") {
            return s
        }
    }
    return path
}
```
